SYoY% = 
VAR CurrentYear = MAX('Calendars'[Date].[Year]) 

VAR LastYear = CurrentYear - 1

VAR PopulationCurrentYear = 

    CALCULATE(

       SUM(Suicide[suicidesno_x]),

        'Calendars'[Date].[Year] = CurrentYear

        )

 VAR PopulationLastYear = 

CALCULATE(

       SUM(Suicide[suicidesno_x]),

        'Calendars'[Date].[Year] = LastYear

        )

     RETURN

    DIVIDE(PopulationCurrentYear - PopulationLastYear, PopulationLastYear,0)

PopYoY% = 

VAR CurrentYear = MAX('Calendars'[Date].[Year]) 

VAR LastYear = CurrentYear - 1

VAR PopulationCurrentYear = 

    CALCULATE(
       SUM(Suicide[population_x]),

        'Calendars'[Date].[Year] = CurrentYear

        )

 VAR PopulationLastYear = 

CALCULATE(

       SUM(Suicide[population_x]),

        'Calendars'[Date].[Year] = LastYear

        )

     RETURN

    DIVIDE(PopulationCurrentYear - PopulationLastYear, PopulationLastYear,0)

M-SYoY% = VAR CurrentYear = MAX('Calendars'[Date].[Year])

VAR PrvYear = CurrentYear - 1

VAR maleSuicideRateCurrentYear = 

    CALCULATE(

        [M-S],

        'Calendars'[Date].[Year] = CurrentYear
    )

VAR maleSuicideRatePreviousYear = 

    CALCULATE(

        [M-S],
        'Calendars'[Date].[Year] = PrvYear
    )

RETURN

DIVIDE(

    maleSuicideRateCurrentYear - maleSuicideRatePreviousYear,

    maleSuicideRatePreviousYear,

    0

)

F-SYoY% = 

VAR CurrentYear = MAX('Calendars'[Date].[Year])

VAR PrvYear = CurrentYear - 1
VAR FemaleSuicideRateCurrentYear = 

    CALCULATE(

        SUM('Suicide'[suicidesno_x]),

        'Suicide'[Sex] = "Female",

        'Calendars'[Date].[Year] = CurrentYear

    )
VAR FemaleSuicideRatePreviousYear = 
    CALCULATE(

        SUM('Suicide'[suicidesno_x]),

        'Suicide'[Sex] = "Female",

        'Calendars'[Date].[Year] = PrvYear
    )
RETURN

DIVIDE(
    FemaleSuicideRateCurrentYear - FemaleSuicideRatePreviousYear,

    FemaleSuicideRatePreviousYear,

    0
)
